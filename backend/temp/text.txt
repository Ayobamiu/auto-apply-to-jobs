export async function getJobsSearch(req: Request, res: Response): Promise<void> {
  const userId = req.userId;
  if (!userId) { res.status(401).json({ error: 'Unauthorized' }); return; }

  await ensureDataTables();

  const query = typeof req.query.query === 'string' ? req.query.query.trim() : '';
  const location = typeof req.query.location === 'string' ? req.query.location.trim() : '';
  const company = typeof req.query.company === 'string' ? req.query.company.trim() : '';
  const page = Math.max(1, parseInt(String(req.query.page ?? '1'), 10) || 1);
  const perPage = Math.min(100, Math.max(1, parseInt(String(req.query.perPage ?? '30'), 10) || 30));
  const offset = (page - 1) * perPage;

  const conditions: string[] = ["site = 'greenhouse'", 'is_active = true'];
  const params: unknown[] = [];
  let paramIdx = 1;

  if (query) {
    conditions.push(`to_tsvector('english', COALESCE(title,'') || ' ' || COALESCE(company,'') || ' ' || COALESCE(location,'')) @@ plainto_tsquery('english', $${paramIdx})`);
    params.push(query);
    paramIdx++;
  }

  if (location) {
    conditions.push(`location ILIKE $${paramIdx}`);
    params.push(`%${location}%`);
    paramIdx++;
  }

  if (company) {
    conditions.push(`company ILIKE $${paramIdx}`);
    params.push(`%${company}%`);
    paramIdx++;
  }

  const where = conditions.join(' AND ');

  const countRes = await pool.query<{ count: string }>(
    `SELECT COUNT(*) as count FROM jobs WHERE ${where}`,
    params,
  );
  const totalCount = parseInt(countRes.rows[0]?.count ?? '0', 10);

  const dataRes = await pool.query<SearchRow>(
    `SELECT site, job_id, title, company, url, location, salary_employment_type,
            company_logo_url, ats, greenhouse_slug, payload, last_seen_at
     FROM jobs
     WHERE ${where}
     ORDER BY last_seen_at DESC NULLS LAST
     LIMIT $${paramIdx} OFFSET $${paramIdx + 1}`,
    [...params, perPage, offset],
  );

  const listings = dataRes.rows.map((r) => ({
    site: r.site,
    jobId: r.job_id,
    url: r.url ?? `https://boards.greenhouse.io/${r.greenhouse_slug ?? 'unknown'}/jobs/${r.job_id}`,
    title: r.title ?? undefined,
    company: r.company ?? undefined,
    location: r.location ?? undefined,
    salaryEmploymentType: r.salary_employment_type ?? undefined,
    companyLogoUrl: r.company_logo_url ?? undefined,
    ats: r.ats ?? undefined,
    greenhouseSlug: r.greenhouse_slug ?? undefined,
    departments: (r.payload as any)?.departments ?? [],
  }));

  res.status(200).json({
    listings,
    totalCount,
    page,
    perPage,
    totalPages: Math.ceil(totalCount / perPage),
  });
}

**Filtering out applied jobs**

This is straightforward — `LEFT JOIN user_job_state ujs ON ujs.job_ref = site || ':' || job_id AND ujs.user_id = $userId` then add `AND (ujs.application_submitted IS NULL OR ujs.application_submitted = false)` to the WHERE. You already have the data, just not the join.

You probably also want to filter `lifecycle_status` — if a user saved a job but didn't apply, should it still show? Worth deciding. Options: hide only `application_submitted = true`, or also hide `lifecycle_status IN ('rejected', 'withdrawn')`.

---

**The clustering problem (same company appearing in bunches)**

This happens because you're ordering by `last_seen_at DESC`. If a company posted 8 jobs recently they all bubble to the top together. Two approaches:

- **Deduplicate by company** — one result per company, with a "3 more at Acme" affordance. Good for discovery but loses jobs.
- **Interleave with row_number** — assign a `ROW_NUMBER() OVER (PARTITION BY company ORDER BY last_seen_at DESC)` then sort by that row number first, then recency. So you get rank-1 from each company before seeing any rank-2. This preserves all jobs but spreads them out naturally.

The second approach is better for your use case since users are looking for any match, not just one per company.

---

**Making search better**

A few problems with the current implementation:

The FTS query uses `plainto_tsquery` which doesn't support partial matching — searching "Eng" won't match "Engineer". Consider `websearch_to_tsquery` instead, which is more forgiving of natural language input and handles multi-word phrases better.

Location search is a pure `ILIKE` with no intelligence — "New York" won't match "NYC" or "New York, NY". You could normalize location on ingest, or add common aliases. More impactful: let users search "remote" and have that match against `salary_employment_type` as well as `location`.

There's no relevance ranking — a job where the title exactly matches the query ranks the same as one where it's buried in the company name. `ts_rank` on the FTS vector would help, and you could boost title matches higher than location matches by building a weighted vector: `setweight(to_tsvector('english', COALESCE(title,'')), 'A') || setweight(to_tsvector('english', COALESCE(company,'')), 'B')`.

The description field exists in the DB but isn't included in the FTS index or search. That's probably intentional for speed, but it means searching for a skill like "React" won't match unless the title says React. Adding description at weight C/D would significantly improve recall.

---

**Better results list — what makes the best UX**

A few things that matter most:

**Personalization signal** — you have a profile with skills, title, location, and work authorization. The search should incorporate these passively. If the user's profile says "Software Engineer" in San Francisco, results should bias toward engineering roles in SF even when the search query is generic. This doesn't require ML — a secondary scoring pass that boosts jobs matching profile title keywords and location is enough.

**Saved vs unsaved state** — show clearly which jobs are saved. Users scan lists and re-encounter jobs they've already seen; showing that state inline prevents wasted clicks.

**Staleness** — `last_seen_at` going stale is a signal a job may be closed. Jobs not seen in 7+ days should either be deprioritized or shown with a "may be expired" indicator.

**Missing data handling** — some rows will have null salary, null company logo, null location. The list should degrade gracefully rather than showing empty slots, which makes the list feel inconsistent and low quality.

**Pagination vs infinite scroll** — for job search, infinite scroll works better than page numbers because users scan vertically and rarely know what "page 3" means. But if you keep pagination, showing total count ("847 jobs") gives users a sense of supply.

1. Filter applied jobs first — most impactful, trivially easy

3. Add `ts_rank` based ordering with title weighting
4. Add saved/lifecycle state to the response so the frontend can show it inline
5. Profile-based boosting once 1-4 are solid
